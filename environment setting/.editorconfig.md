## .editorconfigを入れる

&rarr; インデントなどの余計な差分を消すため

例

    root = true

    [*]

    charset = utf-8

    end_of_line = lf

    indent_size = 2

    indent_style = space

    insert_final_newline = true

    trim_trailing_whitespace = true

    [*.md]

    indent_size = 4

    trim_trailing_whitespace = false
